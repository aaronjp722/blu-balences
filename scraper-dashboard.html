Add Scrape Job

Jobs are saved to your Supabase scraper_jobs table and read dynamically by the agent.

Keyword Place Status Added
1

Create table in Supabase SQL Editor

create table scraper_jobs (
  id         uuid primary key default gen_random_uuid(),
  keyword    text not null,
  place      text not null,
  active     boolean default true,
  created_at timestamptz default now()
);
2

Replace hardcoded JOBS in your agent's main.py

def load_jobs():
    try:
        resp = supabase.table("scraper_jobs").select("*").eq("active", True).execute()
        return [{"keyword": r["keyword"], "place": r["place"]} for r in resp.data]
    except Exception as e:
        log.error(f"Could not load jobs from Supabase: {e}")
        return []

# In run_agent(), change:
#   for job in JOBS:
# to:
#   for job in load_jobs():
